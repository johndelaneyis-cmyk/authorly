<script>
const ta=document.getElementById("book-desc");
const cc=document.getElementById("cc");
ta.addEventListener("input",()=>{cc.textContent=ta.value.length});

let selectedGenre="";
document.querySelectorAll(".genre-chip").forEach(chip=>{
  chip.addEventListener("click",e=>{
    e.preventDefault();
    document.querySelectorAll(".genre-chip").forEach(c=>c.classList.remove("active"));
    chip.classList.add("active");
    selectedGenre=chip.dataset.g;
  });
});
document.querySelector(".genre-chip[data-g=\"\"]").classList.add("active");

const btn=document.getElementById("generate");
const out=document.getElementById("output");
const outBody=document.getElementById("output-body");

btn.addEventListener("click",async()=>{
  const desc=ta.value.trim();
  if(desc.length<30){
    outBody.innerHTML="<div class=\"error\">Please paste at least a paragraph (30+ characters) describing your book — characters, conflict, setting, tone.</div>";
    out.classList.add("visible");
    return;
  }
  btn.disabled=true;
  btn.innerHTML="Reading <span class=\"btn-arrow\">…</span>";
  outBody.innerHTML="<div class=\"loading\">Finding comp titles<span class=\"loading-dot\">.</span><span class=\"loading-dot\">.</span><span class=\"loading-dot\">.</span></div>";
  out.classList.add("visible");
  out.scrollIntoView({behavior:"smooth",block:"nearest"});
  try{
    const res=await fetch("/api/comp",{
      method:"POST",
      headers:{"Content-Type":"application/json"},
      body:JSON.stringify({description:desc,genre:selectedGenre})
    });
    const data=await res.json();
    if(!res.ok){
      outBody.innerHTML="<div class=\"error\">"+(data.error||"Something went wrong. Try again?")+"</div>";
    }else{
      outBody.innerHTML=renderMarkdown(data.text||"");
      attachCopyButtons();
    }
  }catch(err){
    outBody.innerHTML="<div class=\"error\">Connection trouble. Check your internet and try again.</div>";
  }finally{
    btn.disabled=false;
    btn.innerHTML="Find comp titles <span class=\"btn-arrow\">→</span>";
  }
});

function renderMarkdown(md){
  let html=md.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;");
  html=html.replace(/^## (.+)$/gm,"<h2>$1</h2>");
  html=html.replace(/\*\*([^*]+?)\*\*/g,"<strong>$1</strong>");
  html=html.replace(/\*([^*\n]+?)\*/g,"<em>$1</em>");
  const lines=html.split("\n");
  const result=[];
  let inList=false;
  for(const line of lines){
    const m=line.match(/^\s*\d+\.\s+(.+)$/);
    if(m){
      if(!inList){result.push("<ol>");inList=true}
      result.push("<li>"+m[1]+"</li>");
    }else{
      if(inList){result.push("</ol>");inList=false}
      const t=line.trim();
      if(!t)continue;
      if(t.startsWith("<h")||t.startsWith("<ol")||t.startsWith("</")){
        result.push(t);
      }else{
        result.push("<p>"+t+"</p>");
      }
    }
  }
  if(inList)result.push("</ol>");
  return result.join("\n");
}

function attachCopyButtons(){
  const paragraphs=outBody.querySelectorAll("p");
  paragraphs.forEach(p=>{
    const txt=p.innerHTML;
    if(/^<strong>(Short|Medium|Detailed):<\/strong>/.test(txt)){
      const b=document.createElement("button");
      b.className="copy-btn";
      b.textContent="Copy";
      b.onclick=()=>{
        const plain=p.textContent.replace(/^(Short|Medium|Detailed):\s*/,"").replace(/\s*Copy\s*$/,"");
        navigator.clipboard.writeText(plain).then(()=>{
          b.textContent="Copied ✓";
          b.classList.add("copied");
          setTimeout(()=>{b.textContent="Copy";b.classList.remove("copied")},1800);
        });
      };
      p.appendChild(document.createTextNode(" "));
      p.appendChild(b);
    }
  });
}
